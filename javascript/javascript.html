Conditionals

Ternary Conditionals

let result = (number % 2 === 0) ? "Even" : "Odd";

If Conditionals

if (number > 0) {
    console.log("The number is positive!");
} else if (number < 0) {
    console.log("The number is negative!");
} else {
    console.log("The number is zero!");
}

Switch Conditionals

switch (number) {
    case 1:
        console.log("You entered One!");
        break;
    case 2:
        console.log("You entered Two!");
        break;
    default:
        console.log("Invalid number!");
}

Try Conditionals

try {
    console.log("Accessing element 5: " + numbers[5]);
} catch (e) {
    console.log("An error occurred: " + e.message);
} finally {
    console.log("This will always execute");
}

Loops

For Loops

for (let i = 0; i <= 10; i++) {
    if (i == 5) { break; }
    if (i % 2 == 0) { continue; }
    console.log(i);
}

For Of Loops

for (let number of array) {
    console.log(number);
}

For In Loops

for (let key in object) {
    console.log(key);
}

While Loops

while (i <= 10) {
    if (i == 5) { break; }
    if (i % 2 == 0) { continue; }
    console.log(i);
    i++;
}

Do While Loops

do {
    if (i == 5) { break; }
    if (i % 2 == 0) { i++; continue; }
    console.log(i);
    i++;
} while (i <= 10);
