Inheritance of Objects

class Professor extends Person {
      teaches;
      constructor(name, teaches) {
        super(name);
        this.teaches = teaches;
      }
      introduceSelf() {
        console.log(
          `My name is ${this.name}, and I will be your ${this.teaches} professor.`,
        );
      }
      grade(paper) {
        const grade = Math.floor(Math.random() * (5 - 1) + 1);
        console.log(grade);
      }
    }
const walsh = new Professor("Walsh", "Psychology");

Encapsulation of Objects

class Student extends Person {
      #year;
      constructor(name, year) {
        super(name);
        this.#year = year;
      }
      #introduceSelf() {
        console.log(`Hi! I'm ${this.name}, and I'm in year ${this.#year}.`);
      }
      canStudyArchery() {
        return this.#year > 1;
      }
    }
const summers = new Student("Summers", 2);

Promise

Synchronous Promise

function fetchData() {
    return fetch('https://jsonplaceholder.typicode.com/todos/1')
        .then(response => response.json())
        .then(data => {
            console.log('First fetch:', data);
            return fetch('https://jsonplaceholder.typicode.com/todos/2');
        })
        .then(response => response.json())
        .then(data => {
            console.log('Second fetch:', data);
        })
        .catch(error => {
            console.error('Error:', error);
        });
}
fetchData();

Asynchronous Promise

async function fetchData() {
    try {
        let response1 = await fetch('https://jsonplaceholder.typicode.com/todos/1');
        let data1 = await response1.json();
        console.log('First fetch:', data1);

        let response2 = await fetch('https://jsonplaceholder.typicode.com/todos/2');
        let data2 = await response2.json();
        console.log('Second fetch:', data2);
    } catch (error) {
        console.error('Error:', error);
    }
}
fetchData();
